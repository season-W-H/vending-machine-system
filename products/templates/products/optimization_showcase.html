YOLOv8核心优化

单例模式实现

避免重复加载模型，确保全局唯一实例，节省内存资源。 

class YOLOv8ProductRecognition:
    _instance = None
    _lock = threading.Lock()
    
    def __new__(cls, *args, **kwargs):
        if cls._instance is None:
            with cls._lock:
                if cls._instance is None:
                    cls._instance = super().__new__(cls)
        return cls._instance

延迟加载机制

支持按需加载模型，提高系统启动速度。 

# 延迟加载（不自动加载模型）
recognizer = YOLOv8ProductRecognition(auto_load=False)

# 需要时再加载
recognizer._load_model()

# 或自动加载
recognizer = YOLOv8ProductRecognition(auto_load=True)

批量处理功能

新增批量检测方法，支持并行处理多张图像。 

# 批量检测多张图像
results = recognizer.detect_products_batch([

# 结果为列表的列表

使用指南

基本使用

from products.yolov8_integration import \
    YOLOv8ProductRecognition

# 创建识别器
recognizer = YOLOv8ProductRecognition()

# 检测商品
results = recognizer.detect_products(
    "image.jpg",
    confidence_threshold=0.5
)

# 完整识别流程
result = recognizer.recognize_and_calculate(
    "image.jpg"
)

批量处理

# 批量检测
results = recognizer.detect_products_batch([
    "image1.jpg",
    "image2.jpg",
    "image3.jpg"
], confidence_threshold=0.5)

# 处理结果
for i, products in enumerate(results):
    print(f"图像{i+1}:")
    for product in products:
        print(f"  - {product['name']}: ¥{product['price']}")
